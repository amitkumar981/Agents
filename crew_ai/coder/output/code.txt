```python
def fibonacci(n):
    sequence = []
    a, b = 0, 1
    for _ in range(n):
        sequence.append(a)
        a, b = b, a + b
    return sequence

# Generate Fibonacci sequence up to the 50th term
fib_sequence = fibonacci(50)

# Save to a text file
with open('fibonacci.txt', 'w') as f:
    for num in fib_sequence:
        f.write(f'{num}\n')

# Verify the output by reading the file
with open('fibonacci.txt', 'r') as f:
    content = f.read()
print(content)
```
